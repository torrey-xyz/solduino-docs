# Quick Start

Use this minimal sketch to initialize WiFi and query the Solana RPC endpoint.

```cpp
#include <WiFi.h>
#include <solduino.h>

const char* ssid = "YOUR_WIFI_SSID";
const char* password = "YOUR_WIFI_PASSWORD";

RpcClient client(SOLDUINO_DEVNET_RPC);

void setup() {
  Serial.begin(115200);

  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
  }

  if (!client.begin()) {
    Serial.println("RPC init failed");
    return;
  }

  Serial.println(client.getVersion());
  Serial.println(client.getSlot());
}

void loop() {}
```
